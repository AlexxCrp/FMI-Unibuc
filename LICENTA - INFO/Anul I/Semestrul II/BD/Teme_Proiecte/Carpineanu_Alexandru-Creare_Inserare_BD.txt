create table PACIENT(
Cod_Pacient number(4),
Cod_Casa_Asig number(4),
Nume varchar(20),
Prenume varchar(20),
Data_Nastere date,
constraint pacient_Cod_Pacient_PK primary key (Cod_Pacient)
);

create table AFECTIUNE(
Cod_Afectiune number(4),
Denumire varchar(50),
Descriere varchar(50),
constraint afectiune_Cod_Afectiune_PK primary key (Cod_Afectiune)
);

create table AFECTIUNE_PACIENT(
Cod_Afectiune number(4) ,
Cod_Pacient number(4) ,
Severitate varchar(50),
constraint afectiune_pacient_PK primary key(Cod_Afectiune, Cod_Pacient)
);

create table ISTORIC(
Cod_Istoric number(4),
Cod_Pacient number(4),
Data date,
Rezultat varchar(20),
constraint istoric_Cod_Istoric_pk primary key (Cod_Istoric, Cod_Pacient),
constraint istoric_Cod_Pacient_fk foreign key (Cod_Pacient) references pacient(Cod_Pacient)
);

create table PROGRAMARE(
Cod_Prog number(4),
Cod_Pacient number(4),
Data_Prog date,
constraint programare_cod_prog_pk primary key(Cod_Prog, Cod_Pacient),
constraint programare_Cod_Pacient_fk foreign key (Cod_Pacient) references pacient(Cod_Pacient)
);

create table ANGAJAT(
Cod_Angajat number(4),
Nume varchar(20),
Prenume varchar(20),
Salariu Number(6),
constraint angajat_cod_angajat_pk primary key(Cod_Angajat)
);

create table DOCTOR(
Cod_Doctor number(4),
Cod_Angajat number(4),
Data_Term_Studii date,
Specializare Varchar(30),
constraint doctor_cod_doctor_pk primary key(Cod_Doctor),
constraint doctor_cod_angajat_fk foreign key (Cod_Angajat) references Angajat(Cod_Angajat)
);

create table ASISTENT(
Cod_Asistent number(4),
Cod_Doctor number(4),
Cod_Angajat number(4),
Data_Term_Studii date,
constraint asistent_cod_asistent_pk primary key(Cod_Asistent),
constraint asistent_cod_doctor_fk foreign key (Cod_Doctor) references Doctor(Cod_Doctor),
constraint asistent_cod_angajat_fk foreign key (Cod_Angajat) references Angajat(Cod_Angajat)
);

create table INFIRMIER(
Cod_In number(4),
Cod_Asistent number(4),
Cod_Angajat number(4),
Tip_Studii varchar(20),
constraint infirmier_cod_in_pk primary key(Cod_In),
constraint infirmier_cod_asistent_fk foreign key (Cod_Asistent) references Asistent(Cod_Asistent),
constraint infirmier_cod_angajat_fk foreign key (Cod_Angajat) references Angajat(Cod_Angajat)
);

create table CASA_ASIG(
Cod_Casa_Asig number(4),
Data_Alocare_Buget date,
constraint casa_asig_codca_pk primary key(Cod_Casa_Asig)
);

create table DECONTARE(
Cod_Decontare number(4),
Cod_Casa_Asig number(4),
Da_Nu varchar(5),
constraint decontare_cod_decontare_pk primary key(Cod_Decontare),
constraint decontare_Cod_Casa_Asig_fk foreign key (Cod_Casa_Asig) references Casa_Asig(Cod_Casa_Asig)
);

create table TRATAMENT(
Cod_Tratament number(4),
Cod_Afectiune  number(4),
Cod_Doctor number(4),
Cod_Asistent number(4),
Cod_Pacient number(4),
Cod_Decontare number(4),
Durata_Tr number(3),
constraint tratament_pk primary key(Cod_Tratament, Cod_Afectiune),
constraint tratament_cod_doctor_fk foreign key (Cod_Doctor) references Doctor(Cod_Doctor),
constraint tratament_cod_asistent_fk foreign key (Cod_Asistent) references Asistent(Cod_Asistent),
constraint tratament_Cod_Pacient_fk foreign key (Cod_Pacient) references pacient(Cod_Pacient),
constraint tratament_cod_decontare_fk foreign key(Cod_Decontare) references decontare(Cod_Decontare)
);

create table CONSULTATIE(
Cod_Doctor number(4),
Cod_Prog number(4),
Cod_Pacient number(4),
Diagnostic varchar(50),
constraint consultatie_pk primary key(Cod_Doctor, Cod_Prog, Cod_Pacient)
);

create table CLADIRE(
Cod_Cladire number(4),
Nr_Incaperi number(3),
Nr_Etaje number(2),
constraint cladire_cod_cladire_pk primary key(Cod_Cladire)
);

create table CABINET(
Cod_Cabinet number(4),
Cod_Cladire number(4),
Cod_Doctor number(4),
Descriere_Cabinet varchar(30),
Marime_Cabinet number(3),
constraint cabinet_cod_cabinet_pk primary key(Cod_Cabinet, Cod_Cladire),
constraint cabinet_cod_cladire_fk foreign key(Cod_Cladire) references cladire(Cod_cladire)
);
